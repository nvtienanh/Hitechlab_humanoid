215  void setParams(double xi, double vi, double xf, double vf, double dT)
216  {
217  // Calculate constants C and D
218  double C = 6.0*(xf - xi - vi*dT)/(dT*dT);
219  double D = 2.0*(vf - vi)/dT;
220 
221  // Remember the initial conditions
222  m_xi = xi;
223  m_vi = vi;
224 
225  // Calculate the required initial and final accelerations, and the corresponding jerk (slope of the acceleration ramp)
226  m_ai = C - D;
227  double af = 2.0*D - C;
228  m_jerk = (af - m_ai) / dT;
229  }

391  void setParams(double xi, double vi, double xf, double vf, double vm, double a)
392  {
393  // Ensure the acceleration and maximum velocity specifications are positive
394  vm = fabs(vm);
395  a = fabs(a);
396 
397  // Store the spline parameters
398  m_xi = xi;
399  m_vi = vi;
400  m_xf = xf;
401  m_vf = vf;
402 
403  // Calculate further spline parameters
404  m_dx = m_xf - m_xi;
405 
406  // Initialise the spline solution
407  m_soln_exists = false;
408  m_soln_is_trap = false;
409  m_soln_is_trumps = false;
410 
411  // Check the trapezoidal velocity solutions
412  calcTrapSoln( vm, a); // Saturation at vm solution
413  calcTrapSoln(-vm, a); // Saturation at -vm solution
414 
415  // Check the wedge velocity solutions
416  calcWedgeSoln( a, true); // Positive valley solution
417  calcWedgeSoln( a, false); // Negative valley solution
418  calcWedgeSoln(-a, true); // Positive hill solution
419  calcWedgeSoln(-a, false); // Negative hill solution
420 
421  // Calculate a constant velocity fallback solution (should never actually be required)
422  calcConstSoln(vm);
423  }

440  double x(double t) const
441  {
442  // Evaluate the spline position at the required time t
443  if(t < 0.0) return m_xi + m_vi*t;
444  else if(t > m_tf) return m_xf + m_vf*(t - m_tf);
445  else if(!m_soln_exists) return m_xi + m_const_vt*t;
446  else if(m_soln_is_trap)
447  {
448  if(t <= m_trap_t1) return m_xi + t*(m_vi + 0.5*m_trap_ati*t);
449  else if(t <= m_trap_t2) return m_trap_x1 + m_trap_vms*(t - m_trap_t1);
450  else
451  {
452  double tb = m_tf - t;
453  return m_xf - tb*(m_vf - 0.5*m_trap_atf*tb);
454  }
455  }
456  else
457  {
458  if(t <= m_wedge_t3) return m_xi + t*(m_vi - 0.5*m_wedge_as*t);
459  else
460  {
461  double tb = m_tf - t;
462  return m_xf - tb*(m_vf - 0.5*m_wedge_as*tb);
463  }
464  }
465  }

468  double v(double t) const
469  {
470  // Evaluate the spline velocity at the required time t
471  if(t < 0.0) return m_vi;
472  else if(t > m_tf) return m_vf;
473  else if(!m_soln_exists) return m_const_vt;
474  else if(m_soln_is_trap)
475  {
476  if(t <= m_trap_t1) return m_vi + m_trap_ati*t;
477  else if(t <= m_trap_t2) return m_trap_vms;
478  else return m_vf - m_trap_atf*(m_tf - t);
479  }
480  else
481  {
482  if(t <= m_wedge_t3) return m_vi - m_wedge_as*t;
483  else return m_vf - m_wedge_as*(m_tf - t);
484  }
485  }

488  double a(double t) const
489  {
490  // Evaluate the spline acceleration at the required time t
491  if(t < 0.0 || t > m_tf || !m_soln_exists) return 0.0;
492  else if(m_soln_is_trap)
493  {
494  if(t <= m_trap_t1) return m_trap_ati;
495  else if(t <= m_trap_t2) return 0.0;
496  else return m_trap_atf;
497  }
498  else
499  {

501  else return m_wedge_as;
502  }

525  // Trapezoidal velocity solution calculation
526  void calcTrapSoln(double vms, double a) // vms should be +-vm, a should be +ve
527  {
528  // Calculate the required timing keypoints
529  double ti1 = fabs(vms - m_vi) / a;
530  double t2f = fabs(vms - m_vf) / a;
531  double dxi1 = 0.5 * ti1 * (vms + m_vi);
532  double dx2f = 0.5 * t2f * (vms + m_vf);
533  double t12 = (m_dx - dxi1 - dx2f) / vms;
534 
535  // Return if the solution is invalid
536  if(t12 < 0.0) return;
537 
538  // Calculate additional parameters
539  double ti2 = ti1 + t12;
540  double tif = ti2 + t2f;
541 
542  // Update the currently chosen solution if this solution is better
543  if(!m_soln_exists || !m_soln_is_trumps || (tif <= m_tf))
544  {
545  m_trap_ati = a * sign0(vms - m_vi);
546  m_trap_atf = a * sign0(m_vf - vms);
547  m_trap_vms = vms;
548  m_trap_t1 = ti1;
549  m_trap_t2 = ti2;
550  m_trap_x1 = m_xi + dxi1;
551  m_tf = tif;
552  m_soln_exists = true;
553  m_soln_is_trap = true;
554  m_soln_is_trumps = true;
555  }
556  }

558  // Wedge velocity solution calculation
559  void calcWedgeSoln(double as, bool soln) // as should be +-a, soln is a flag for whether to calculate the positive or negative solution
560  {
561  // Calculate the wedge solution discriminant
562  double discr = 0.5*(m_vi*m_vi + m_vf*m_vf) - as*m_dx;
563 
564  // Return if there is no solution
565  if(discr < 0.0) return;
566 
567  // Calculate the corner time and return if it is invalid
568  double t3 = (soln ? m_vi + sqrt(discr) : m_vi - sqrt(discr)) / as;
569  if(t3 < 0.0) return;
570 
571  // Calculate the total time and return if it is invalid
572  double tf = 2.0*t3 + (m_vf - m_vi) / as;
573  if(tf < t3) return;
574 
575  // Calculate whether the solution is trumps
576  double v3 = m_vi - as*t3;
577  bool is_trumps = (fabs(v3) <= fabs(m_vi)) && (v3 * m_vi >= 0.0);
578 
579  // Update the currently chosen solution if this solution is better
580  bool soln_is_shorter = (tf <= m_tf);
581  if(!m_soln_exists || (m_soln_is_trumps ? is_trumps && soln_is_shorter : is_trumps || soln_is_shorter))
582  {
583  m_wedge_as = as;
584  m_wedge_t3 = t3;
585  m_tf = tf;
586  m_soln_exists = true;
587  m_soln_is_trap = false;
588  m_soln_is_trumps = is_trumps;
589  }
590  }

596  if(!m_soln_exists)
597  {

600  }
