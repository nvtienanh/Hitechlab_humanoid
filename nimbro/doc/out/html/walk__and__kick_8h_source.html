233  // TheWorm class (https://en.wikipedia.org/wiki/Worm_%28marketing%29)
234  class TheWorm
235  {
236  public:
237  explicit TheWorm(int limit = 100) { reset(limit); }
238  void reset() { m_worm = 0; }
239  void reset(int limit) { reset(); setLimit(limit); }
240  void setLimit(int limit) { m_limit = (limit >= 1 ? limit : 1); } // The count from zero to a positive or negative decision
241  void setRange(int range) { m_limit = (range >= 2 ? range >> 1 : 1); } // The count from a decision to the opposite decision
242  void setWorm(int value) { m_worm = (value >= m_limit ? m_limit : (value <= -m_limit ? -m_limit : value)); }
243  void vote(bool decision) { if (decision) m_worm = (m_worm < m_limit ? m_worm + 1 : m_limit); else m_worm = (m_worm > -m_limit ? m_worm - 1 : -m_limit); }
244  bool decision() const { return (m_worm >= 0); } // Balanced votes is a decision of true!
245  bool unanimousTrue() const { return (m_worm >= m_limit); }
246  bool unanimousFalse() const { return (m_worm <= -m_limit); }
247  bool unanimous() const { return (unanimousTrue() || unanimousFalse()); }
248  private:
249  int m_limit;
250  int m_worm;
251  };

258  void reset() { m_valid = false; m_x = m_v = m_t = 0.0; }

264  void newTarget(double x, double v, double maxVel, double maxAcc, bool ctsVel = true)
265  {
266  m_spline.setParams(m_x, (ctsVel ? m_v : 0.0), x, v, maxVel, maxAcc);
267  m_valid = true;
268  m_t = 0.0;
269  }
270  double forward(double dT)
271  {
272  if (!m_valid) return m_x;
273  m_t += dT;
274  m_x = m_spline.x(m_t);
275  m_v = m_spline.v(m_t);
276  return m_x;
277  }
