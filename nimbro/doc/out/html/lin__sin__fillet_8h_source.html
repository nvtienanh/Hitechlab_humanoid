33  void reset() { m_A = m_B = m_a = m_b = m_c = m_tPre = m_tPost = 0.0; }

35  // Update function
36  bool update(double A, double B, double T)
37  {
38  double BT = B*T;
39  if(T < 1e-6 || fabs(B) < 1e-6 || fabs(BT) > M_PI_2 - 1e-6)
40  {
41  reset();
42  return false;
43  }
44  m_A = A;
45  m_B = B;
46  if(fabs(A) < 1e-6)
47  {
48  m_a = m_b = m_c = 0.0;
49  m_tPre = m_tPost = T;
50  return true;
51  }
52  m_c = m_A*sin(BT);
53  m_b = m_A*m_B*cos(BT);
54  m_a = m_b*m_b/(4*m_c);
55  m_tPre = T - 2*m_c/m_b;
56  m_tPost = T;
57  if(m_tPre > m_tPost) m_tPre = m_tPost; // Should never happen
58  return true;
59  }
60 
61  // Update functions with maximum allowed pre-time
62  bool update(double A, double B, double T, double maxPre) // maxPre should be non-negative and be the largest acceptable pre-time (time from start of fillet to the mid-fillet lin-sin transition)
63  {
64  if(!update(A, B, T)) return false; // Calculate the fillet that results from using time T
65  if(maxPre < 1e-6)
66  {
67  reset(); // Disable the fillet if maxPre is too small
68  return true;
69  }
70  if(m_tPre >= -maxPre) return true;
71  return update(A, B, T*(maxPre/-m_tPre)); // Reduce T to (conservatively) abide by the maxPre specification
72  }
73  bool updateExact(double A, double B, double T, double maxPre)
74  {
75  if(!update(A, B, T)) return false; // Calculate the fillet that results from using time T
76  if(maxPre < 1e-6)
77  {
78  reset(); // Disable the fillet if maxPre is too small
79  return true;
80  }
81  if(m_tPre >= -maxPre) return true;
82  double newT = T;
83  int count = 0;
84  while(true)
85  {
86  if(fabs(m_tPre + maxPre) < 1e-6) return true;
87  if(fabs(B*newT) > M_PI_2 - 1e-6) break;
88  double secBT = 1 / cos(B*newT);
89  newT -= (m_tPre + maxPre) / (1 - 2*secBT*secBT);
90  if(!update(A, B, newT)) break;
91  if(++count >= 20) break;
92  }
93  return update(A, B, T, maxPre);
94  }

96  // Return whether this fillet needs to be applied
97  bool hasEffect(double t) const { return (t > m_tPre && t < m_tPost); }
98 
99  // Evaluate function (evaluates just the required waveform to add to get the fillet)
100  double eval(double t) const
101  {
102  if(!hasEffect(t)) return 0.0;
103  double dt = t - m_tPost;
104  double parabola = m_c + dt*(m_b + dt*m_a);
105  if(t <= 0.0) return parabola;
106  else return parabola - m_A*sin(m_B*t);
107  }
